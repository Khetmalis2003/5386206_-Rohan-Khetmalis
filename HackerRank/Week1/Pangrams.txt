char* pangrams(char* s) {
    int seen[26];
    for (int i = 0; i < 26; i++) seen[i] = 0;
    for (char *p = s; *p != '\0'; p++) {
        char ch = *p;
        if (ch >= 'A' && ch <= 'Z') ch = ch - 'A' + 'a';
        if (ch >= 'a' && ch <= 'z') {
            seen[ch - 'a'] = 1;
        }
    }
    for (int i = 0; i < 26; i++) {
        if (seen[i] == 0) {
            static char no[] = "not pangram";
            return no;
        }
    }
    static char yes[] = "pangram";
    return yes;
}
